  3     /**

  5     * @memberOf light8.util.collection

  9     function size(collection) {
 10         if (!collection || typeof collection !== 'object') {
 11             return 0;
 12         }
 13 
 14         return collection.length;
 15     }
 16 
 17     /**
 18      * Remove an item out of collection by index
 19      * @memberOf light8.util.collection
 20      * @method removeByIndex
 21      * @param {Object} collection The collection
 22      * @param {Number} index The index of item
 23      */
 24     function removeByIndex(collection, index) {
 25         if (!collection) {
 26             return;
 27         }
 28 
 29         collection.splice(index, 1);
 30     }
 31 
 32     /** @namespace light8.util.collection */
 33     light8.util.collection = {
