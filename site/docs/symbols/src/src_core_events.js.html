1 (function (namespace) {
  2 
  3     "use strict";
  4 
  5     var collectionUtil = light8.util.collection;
  6 
  7     if (light8.Event) {
  8         return;
  9     }
 10 
 11     /**
 12      * The event class of light8.
 13      * @class Event
 14      * @memberOf light8
 15      */
 16     light8.Event = fabric.util.createClass(/** @scope light8.Event.prototype */ {
 17 
 18         /**
 19          * The Object container all events and its action
 20          * @private
 21          * @type Object
 22          */
 23         _eventsContainer: {},
 24 
 25         /**
 26          * Bind a function to an event
 27          * @method bind
 28          * @param {String} eventName Event's Name
 29          * @param {Function} func Function do event action
 30          */
 31         bind: function (eventName, func) {
 32             var eventContainer = this._eventsContainer[eventName];
 33 
 34             // initialize an event container if not exist
 35             if (!eventContainer) {
 36                 this._eventsContainer[eventName] = eventContainer = [];
 37             }
 38 
 39             // add event to event container
 40             eventContainer.push(func);
 41         },
 42 
 43         /**
 44          * Unbind a function out of an event
 45          * @method unbind
 46          * @param {String} eventName Event's Name
 47          * @param {Function} func Function do event action
 48          */
 49         unbind: function (eventName, func) {
 50             this._doEachEvent(eventName, function (container, index) {
 51                 if (container[index] === func) {
 52                     collectionUtil.removeByIndex(container, index); // remove function out of a event
 53                 }
 54             });
 55         },
 56 
 57         /**
 58          * Clean an event
 59          * @method clean
 60          * @param {String} eventName Event's Name
 61          */
 62         clean: function (eventName) {
 63             delete this._eventsContainer[eventName];
 64         },
 65 
 66         /**
 67          * Clean all event of this object
 68          * @method reset
 69          */
 70         reset: function () {
 71             this._eventsContainer = {};
 72         },
 73 
 74         /**
 75          * Trigger an event
 76          * @method trigger
 77          * @param {String} eventName Event's Name
 78          * @param {Any[]} [...] Values to pass to a bound function
 79          */
 80         trigger: function (eventName) {
 81             var args = arguments,
 82                 self = this;
 83 
 84             // with each func in event container, apply it with arguments pass to trigger function
 85             this._doEachEvent(eventName, function (container, index) {
 86                 container[index].apply(self, Array.prototype.slice.call(args, 1));
 87             });
 88         },
 89 
 90         /**
 91          *
 92          * @param func
 93          */
 94         _doEachEvent: function (eventName, func) {
 95             var eventContainer = this._eventsContainer[eventName],
 96                 eventLength;
 97 
 98             // initialize an event container if not exist
 99             if (eventContainer) {
100                 eventLength = collectionUtil.size(eventContainer);
101 
102                 for (var i = eventLength; i >=0; i--) {
103                     // call always fast than apply
104                     func.call(this, eventContainer, i);
105                 }
106             }
107         }
108 
109     });
110 
111 }).call(this, light8);
